What is Lambda?

A lambda is a small anonymous function — perfect for short, one-time operations especially with sort() and filter().

python
# Regular function def double(x): return x * 2 # Equivalent lambda double = lambda x: x * 2 print(double(5)) # 10 # Lambda with multiple args add = lambda x, y: x + y print(add(3, 4)) # 7 # Lambda with condition classify = lambda x: "High" if x > 50000 else "Low" print(classify(72000)) # High
Output
10 7 High

Lambda with sorted() and map()

Inheritance

class Animal: def __init__(self, name, sound): self.name = name self.sound = sound def speak(self): return f"{self.name} says {self.sound}!" class Dog(Animal): # Inherits from Animal def fetch(self): return f"{self.name} fetches the ball!" class Cat(Animal): def speak(self): # Override parent method return f"{self.name} says Meow (quietly)..." dog = Dog("Rex", "Woof") cat = Cat("Luna", "Meow") print(dog.speak()) # Inherited print(dog.fetch()) # Dog-specific print(cat.speak()) # Overridden
Output
Rex says Woof! Rex fetches the ball! Luna says Meow (quietly)...

class Manager(Employee): def __init__(self, name, dept, salary, team_size): super().__init__(name, dept, salary) # Call parent __init__ self.team_size = team_size def __str__(self): # String representation return f"Manager {self.name} ({self.team_size} reports)" def __repr__(self): # Developer representation return f"Manager('{self.name}','{self.dept}',{self.salary})" def __len__(self): # len() support return self.team_size mgr = Manager("Layla","Sales",18000,8) print(mgr) # calls __str__ print(len(mgr)) # calls __len__ → 8
