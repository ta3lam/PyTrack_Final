📈 Matplotlib Basics

Unit 8 — Turn numbers into insights — create professional charts with Matplotlib, Seaborn, and Plotly.

Your First Chart

python
import matplotlib.pyplot as plt months = ["Jan","Feb","Mar","Apr","May","Jun"] sales = [45000,52000,48000,61000,58000,72000] fig, ax = plt.subplots(figsize=(10, 5)) ax.plot(months, sales, marker="o", linewidth=2.5, color="#3b82f6", markersize=8) ax.set_title("Monthly Sales 2024", fontsize=15, fontweight="bold", pad=15) ax.set_xlabel("Month", fontsize=12) ax.set_ylabel("Sales (EGP)", fontsize=12) ax.grid(True, alpha=0.3, linestyle="--") ax.spines["top"].set_visible(False) ax.spines["right"].set_visible(False) plt.tight_layout() plt.savefig("sales_chart.png", dpi=150, bbox_inches="tight") plt.show()

Figure Anatomy

python
# Figure = the whole canvas # Axes = the plot area (can have multiple per figure) # Axis = x-axis or y-axis fig, ax = plt.subplots(figsize=(10,5)) # Annotations ax.plot(months, sales, "bo-") ax.annotate("Peak!", xy=(5,72000), xytext=(4,68000), arrowprops=dict(arrowstyle="->",color="red"), color="red", fontsize=11) # Add horizontal reference line ax.axhline(y=sum(sales)/len(sales), color="orange", linestyle="--", label=f"Avg: {sum(sales)/len(sales):,.0f}") ax.legend() plt.show()
💡 Tip: Always use fig, ax = plt.subplots() — gives full control and handles multiple subplots easily.

import matplotlib.pyplot as plt import numpy as np regions = ["Cairo","Alex","Giza","Port Said","Aswan"] sales = [450000,280000,195000,120000,87000] colors = ["#3b82f6","#06b6d4","#8b5cf6","#f59e0b","#10b981"] fig, ax = plt.subplots(figsize=(10,6)) bars = ax.bar(regions, sales, color=colors, width=0.6, edgecolor="white", linewidth=0.5) # Value labels on bars for bar, val in zip(bars, sales): ax.text(bar.get_x() + bar.get_width()/2, val + 8000, f"{val/1000:.0f}K", ha="center", va="bottom", fontsize=10, fontweight="bold") ax.set_title("Sales by Region — Q4 2024", fontsize=14, fontweight="bold") ax.set_ylabel("Sales (EGP)") ax.yaxis.set_major_formatter(plt.FuncFormatter(lambda x,_: f"{x/1000:.0f}K")) ax.grid(axis="y", alpha=0.3, linestyle="--") ax.spines["top"].set_visible(False); ax.spines["right"].set_visible(False) plt.tight_layout(); plt.show()

📈 Histograms & Scatter Plots

Unit 8 — Turn numbers into insights — create professional charts with Matplotlib, Seaborn, and Plotly.

Histograms — Distribution

python
import matplotlib.pyplot as plt import numpy as np np.random.seed(42) salaries = np.random.normal(9000, 2000, 500) fig, axes = plt.subplots(1, 2, figsize=(14, 5)) # Basic histogram axes[0].hist(salaries, bins=30, color="#3b82f6", alpha=0.7, edgecolor="white") axes[0].axvline(salaries.mean(), color="red", linestyle="--", linewidth=2, label=f"Mean: {salaries.mean():,.0f}") axes[0].axvline(np.median(salaries), color="green", linestyle="--", linewidth=2, label=f"Median: {np.median(salaries):,.0f}") axes[0].set_title("Salary Distribution"); axes[0].legend() # KDE (smooth density) from scipy.stats import gaussian_kde kde = gaussian_kde(salaries) x = np.linspace(salaries.min(), salaries.max(), 300) axes[1].plot(x, kde(x), color="#3b82f6", linewidth=2) axes[1].fill_between(x, kde(x), alpha=0.3, color="#3b82f6") axes[1].set_title("KDE (Smooth Distribution)") plt.tight_layout(); plt.show()

import matplotlib.pyplot as plt, numpy as np from scipy.stats import gaussian_kde, ttest_ind np.random.seed(42) before = np.random.normal(45000, 10000, 100) after = np.random.normal(52000, 11000, 100) fig, axes = plt.subplots(1, 2, figsize=(14,5)) fig.suptitle("Sales Distribution: Before vs After Campaign", fontsize=14, fontweight="bold") for ax, data, label, color in zip( axes, [before, after], ["Before","After"], ["#ef4444","#10b981"] ): ax.hist(data, bins=20, color=color, alpha=0.5, edgecolor="white", density=True) x = np.linspace(data.min(), data.max(), 200) ax.plot(x, gaussian_kde(data)(x), color=color, linewidth=2) ax.axvline(data.mean(), color="black", linestyle="--", label=f"Mean: {data.mean():,.0f}") ax.set_title(f"{label} Campaign") ax.set_xlabel("Sales (EGP)"); ax.legend() t, p = ttest_ind(before, after) print(f"T-test p-value: {p:.4f}") print(f"Mean increase: {after.mean()-before.mean():,.0f} ({(after.mean()/before.mean()-1)*100:.1f}%)") plt.tight_layout(); plt.show()

Pie Charts

python
import matplotlib.pyplot as plt products = ["Laptop","Phone","Tablet","Monitor","Other"] sales = [45, 30, 15, 8, 2] colors = ["#3b82f6","#06b6d4","#8b5cf6","#f59e0b","#94a3b8"] explode = [0.05, 0.05, 0, 0, 0] # Slightly pull out top slices fig, axes = plt.subplots(1, 2, figsize=(14, 6)) # Pie chart axes[0].pie(sales, labels=products, colors=colors, explode=explode, autopct="%1.1f%%", pctdistance=0.85, startangle=90, wedgeprops=dict(edgecolor="white", linewidth=2)) axes[0].set_title("Market Share by Product") # Donut chart (pie with hole) axes[1].pie(sales, colors=colors, autopct="%1.1f%%", startangle=90, wedgeprops=dict(edgecolor="white",width=0.6)) axes[1].set_title("Donut Chart") centre = plt.Circle((0,0), 0.40, fc="white") axes[1].add_patch(centre) plt.tight_layout(); plt.show()

📈 Seaborn Charts

Unit 8 — Turn numbers into insights — create professional charts with Matplotlib, Seaborn, and Plotly.

Why Seaborn?

Seaborn creates beautiful statistical charts with minimal code. Built on Matplotlib with smart defaults.

python
import seaborn as sns, matplotlib.pyplot as plt, pandas as pd, numpy as np np.random.seed(42) sns.set_theme(style="whitegrid", palette="muted") df = pd.DataFrame({ "Sales": np.random.normal(50000,15000,200), "Region": np.random.choice(["Cairo","Alex","Giza"],200), "Product":np.random.choice(["Laptop","Phone","Tablet"],200), "Quarter":np.random.choice(["Q1","Q2","Q3","Q4"],200), }) fig, axes = plt.subplots(2, 2, figsize=(14,10)) # Boxplot — distribution + outliers sns.boxplot(data=df, x="Region", y="Sales", ax=axes[0,0], palette="Blues") axes[0,0].set_title("Sales Distribution by Region") # Violin plot — distribution shape sns.violinplot(data=df, x="Product", y="Sales", ax=axes[0,1], palette="Set2") axes[0,1].set_title("Sales Distribution by Product") # Strip/swarm plot — individual points sns.stripplot(data=df, x="Region", y="Sales", ax=axes[1,0], palette="Set1", alpha=0.5, jitter=True) axes[1,0].set_title("Individual Sales Points") # Bar plot (with confidence intervals) sns.barplot(data=df, x="Quarter", y="Sales", ax=axes[1,1], palette="rocket", errorbar="sd") axes[1,1].set_title("Average Sales by Quarter") plt.tight_layout(); plt.show()
💡 Tip: Seaborn themes: 'darkgrid', 'whitegrid', 'dark', 'white', 'ticks'. Palettes: 'Blues', 'Set2', 'rocket', 'muted', 'deep'.
